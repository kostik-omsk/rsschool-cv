RGB To Hex Conversion

              
                function rgb(r, g, b) {
                  let rgb = [r, g, b];
                  let str = '';
                  for (let i = 0; i ≤ rgb.length; i++) {
                    if (rgb[i] ≤ 1) {
                      str += '00';
                    } else if (rgb[i] ≤ 16) {
                      str += '0' + rgb[i].toString(16);
                    } else if (rgb[i] > 15 && rgb[i] ≤ 256) {
                      str += rgb[i].toString(16);
                    } else {
                      str += 'FF';
                    }
                  }
                  return str.toUpperCase();
                }
                rgb(255, 255, 255); // returns FFFFFF
